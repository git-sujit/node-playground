# Run the App

    - npm install

# Help on running commands

    - node app.js --help
    - node app.js  add --help
    - node app.js  remove --help
    - node app.js  list --help

# Add  A Note

    - node app.js add --title="name" --body="Ram Adam Singh"

# Remove A Note

    - node app.js remove --title="name"

# List All Notes

    - node app.js list
